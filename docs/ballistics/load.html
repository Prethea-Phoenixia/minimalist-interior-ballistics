 20def mark_max_pressure(state_generating_func: Callable) -> Callable:
 21    """
 22    decorator that finds the maximum pressure point and insert it into the list of
 23    `ballistics.state.State` sorted by time, given a list-of-`ballistics.state.State`
 24    generating function.
 25
 26    Notes
 27    -----
 28    Implementation wise, conducts a gold-section search using `ballistics.gss.gss`
 29    in the interval bracketed by the step before and after the step where maximum
 30    pressure is recorded. By intercepting the accuracy specification passed to the
 31    generating functions, the peak-pressure point is determined to (at worst) step size
 32    times `acc`.
 33    """
 34
 35    @wraps(state_generating_func)
 36    def wrapper(self, *args, acc, **kwargs):
 37        states = state_generating_func(self, *args, acc=acc, **kwargs)
 38
 39        pressures = [s.average_pressure for s in states]
 40        j = pressures.index(max(pressures))
 41        s_pmax = states[j]
 42
 43        i, k = max(j - 1, 0), min(j + 1, len(states) - 1)
 44        time_min, time_max = states[i].time, states[k].time
 45
 46        def time_pressure(time: float) -> float:
 47            return self.propagate_rk4(
 48                state=s_pmax, dt=time - s_pmax.time
 49            ).average_pressure
 50
 51        time_pmax = gss(
 52            f=time_pressure,
 53            x_0=time_min,
 54            x_1=time_max,
 55            find=FIND_MAX,
 56            tol=acc * (time_max - time_min) / (k - i),
 57        )
 58
 59        s_pmax = self.propagate_rk4(
 60            state=s_pmax, dt=time_pmax - s_pmax.time, marker=PEAK_PRESSURE
 61        )
 62        insort(states, s_pmax)
 63
 64        return states
 65
 66    return wrapper

 90    def add_charge(self, *args, **kwargs):
 91        self.charges.append(Charge(load=self, *args, **kwargs))
 92        self.total_charge_mass = sum(c.charge_mass for c in self.charges)
 93
 94        # calculate the average adiabatic index
 95        molar_sum = sum(c.charge_mass / c.gas_molar_mass for c in self.charges)
 96        average_Cp = 0
 97        average_Cv = 0
 98        for c in self.charges:
 99            adiabatic_index = c.adiabatic_index
100            molar_fraction = (c.charge_mass / c.gas_molar_mass) / molar_sum
101            average_Cp += molar_fraction * (adiabatic_index) / (adiabatic_index - 1)
102            average_Cv += molar_fraction / (adiabatic_index - 1)
103
104        self.average_adiabatic_index = average_Cp / average_Cv
105
106        # calculation variables
107        self.phi = (
108            1 + self.loss_fraction + self.total_charge_mass / (3 * self.shot_mass)
109        )

112    def to_burnout(self, n_intg: int, acc: float) -> List[State]:
113        """integrates projectile motion up to the propellant burnout point and returns
114        a List of State.
115
116        Parameters
117        ----------
118        n_intg: int
119            number of integration steps that is to be taken from shot start to charge
120            burnout.
121
122        acc: float
123            relative accuracy characteristic points (burnout and peak) will be
124            determined to in relation to actual step size.
125
126        Returns
127        -------
128        list of `ballistics.state.State`
129
130        Notes
131        -----
132        Implementation wise, the adaptive stepsize is first seeded with a
133        value from module wide constant `ballistics.MAX_DT`, integrated until burnout,
134        the total time of which is used as a better approximate to time-to-burnout,
135        divided by `n_intg` as the step size for the next run. This is repeated
136        until at least `n_intg` steps were taken. This ensures the final two steps
137        brackets the actual burnout point, from which `ballistics.dekker.dekker`
138        is called to numerically find the burnout point to an accuracy of stepsize times
139        `acc`.
140        """
141        Z_c0 = tuple(c.solve_bomb() for c in self.charges)
142
143        def burnout(state: State) -> float:
144            if sum(Z - c.Z_k for Z, c in zip(state.burnup_fractions, self.charges)) < 0:
145                return -1
146            else:
147                return 1
148
149        states: List[State] = []
150        delta_t = MAX_DT
151        rough_ttb = 0.0
152        while len(states) < n_intg:
153            if rough_ttb > 0:
154                delta_t = rough_ttb / n_intg
155            states = []
156            s_next = State(
157                load=self,
158                time=0,
159                travel=0,
160                velocity=0,
161                burnup_fractions=Z_c0,
162                marker=START,
163            )
164            while burnout(s_next) < 0:
165                s_now = s_next
166                states.append(s_now)
167                s_next = self.propagate_rk4(state=s_now, dt=delta_t)
168
169            rough_ttb = s_next.time
170
171        tol_t = delta_t * acc
172
173        def time_burnout(time: float) -> float:
174            return burnout(self.propagate_rk4(state=s_now, dt=time - s_now.time))
175
176        burnout_time = dekker(
177            f=time_burnout, x_0=s_now.time, x_1=s_next.time, tol=tol_t
178        )[0]
179        s_burnout = self.propagate_rk4(
180            state=s_now, dt=burnout_time - s_now.time, marker=BURNOUT
181        )
182        states.append(s_burnout)
183        return states

185    def to_travel(self, travel: float, n_intg: int, acc: float) -> List[State]:
186        """conducts integration up to the desired shot-travel using length-wise ODE
187        from burnout point to muzzle exit. Calls `.to_burnout` for integration up
188        to the burnout point.
189        Parameters
190        ----------
191        travel: float
192            the projectile travel to which the integration is done to. Does not accept
193            a value that results in muzzle exit before burnout.
194        n_intg: int
195            determines the number of steps taken in the length-wise integration,
196            in addition to being passed through to `to_burnout`.
197        acc: float
198            see documentation for `to_burnout`.
199
200        Returns
201        -------
202        list of `ballistics.state.State`
203
204        """
205        states = self.to_burnout(n_intg=n_intg, acc=acc)
206        s_burnout = states[-1]
207
208        d_travel = (travel - s_burnout.travel) / n_intg
209
210        if d_travel < 0:
211            raise ValueError("projectile unable to achieve burnout at specified travel")
212
213        state = s_burnout
214        for _ in range(n_intg - 1):
215            states.append(state := self.propagate_rk4(state=state, dl=d_travel))
216
217        states.append(
218            state := self.propagate_rk4(state=state, dl=d_travel, marker=MUZZLE)
219        )
220
221        return states

264    def dt(self, state: State) -> Delta:
265        P = state.average_pressure
266        dZ = tuple(
267            c.dZdt(P) if Z < c.Z_k else 0
268            for Z, c in zip(state.burnup_fractions, self.charges)
269        )
270        return Delta(
271            d_time=1,
272            d_travel=state.velocity,
273            d_velocity=self.S * P / (self.phi * self.shot_mass),
274            d_burnup_fractions=dZ,
275        )

277    def dl(self, state: State) -> Delta:
278        # d/dl = d/dt * dt/dl
279        # dt / dl = 1/v
280        v = state.velocity
281        dt = self.dt(state)
282
283        return dt / v
284
285    def dv(self, state: State) -> Delta:
286        # d / dv = d/dt * dt/dv
287        dt = self.dt(state)
288        return dt / dt.d_velocity

290    def propagate_rk4(
291        self, state: State, dt=..., dl=..., dv=..., marker: str = STEP
292    ) -> State:
293        s_i = state.increment
294
295        if dt != ...:
296            df, dx, d_arg = self.dt, dt, "dt"
297        elif dl != ...:
298            df, dx, d_arg = self.dl, dl, "dl"
299        elif dv != ...:
300            df, dx, d_arg = self.dv, dv, "dv"
301        else:
302            raise ValueError("at least one of `dv, dt, dl` must be specified.")
303
304        def generate_dargs(value: float) -> Dict[str, float]:
305            return {d_arg: value}
306
307        k1 = df(state)
308        k2 = df(s_i(d=0.5 * k1 * dx, **generate_dargs(0.5 * dx), marker=INTERMEDIATE))
309        k3 = df(s_i(d=0.5 * k2 * dx, **generate_dargs(0.5 * dx), marker=INTERMEDIATE))
310        k4 = df(s_i(d=k3 * dx, **generate_dargs(dx), marker=INTERMEDIATE))
311        return s_i(
312            d=(k1 + k2 * 2 + k3 * 2 + k4) * dx / 6, **generate_dargs(dx), marker=marker
313        )

145            if sum(Z - c.Z_k for Z, c in zip(state.burnup_fractions, self.charges)) < 0:
146                return -1
147            else:
148                return 1
149
150        states: List[State] = []

152        rough_ttb = 0.0
153        while len(states) < n_intg:
154            if rough_ttb > 0:
155                delta_t = rough_ttb / n_intg
156            states = []

159                time=0,

167                states.append(s_now)
168                s_next = self.propagate_rk4(state=s_now, dt=delta_t)
169
170            rough_ttb = s_next.time
171

173

179        )[0]

181            state=s_now, dt=burnout_time - s_now.time, marker=BURNOUT
182        )
183        states.append(s_burnout)
