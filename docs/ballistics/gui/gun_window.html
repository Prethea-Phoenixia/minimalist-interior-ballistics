143    def define_gun(self):
144        try:
145            name, family = (e.get() for e in self.value_entries[:2])
146            (
147                cross_section,
148                shot_mass,
149                charge_mass,
150                chamber_volume,
151                loss_fraction,
152                start_pressure,
153                reduced_burnrate,
154                travel,
155            ) = (float(e.get()) for e in self.value_entries[2:])
156
157            cross_section *= 1e-2  # dm^2 to m^2
158            chamber_volume *= 1e-3  # L to m^3
159            start_pressure *= 1e6  # MPa to Pa
160            loss_fraction *= 1e-2  # % to 1
161            reduced_burnrate *= 1e-6  # μs to s
162            travel *= 1e-1
163
164            prop = self.get_prop()
165            ff = self.form_function_frame.get_form_function()
166            # the charge name is automatically generated.
167            charge = Charge.from_propellant(
168                name=" ".join((prop.name, ff.name)),
169                description=ff.description,
170                family=family,
171                reduced_burnrate=reduced_burnrate,
172                propellant=prop,
173                form_function=ff,
174            )
175
176            self.gun = Gun(
177                name=name,
178                description=self.text.get("1.0", "end-1c"),
179                cross_section=cross_section,
180                shot_mass=shot_mass,
181                charge=charge,
182                charge_mass=charge_mass,
183                chamber_volume=chamber_volume,
184                loss_fraction=loss_fraction,
185                start_pressure=start_pressure,
186                travel=travel,
