 82    def get_prop(self) -> Optional[Propellant]:
 83        i = self.prop_combo.current()
 84        if i == -1:
 85            raise ValueError("no propellant has been selected.")
 86        else:
 87            return self.props[i]

 89    def define_gun(self):
 90        try:
 91            name = self.value_entry[0].get()
 92            (
 93                cross_section,
 94                shot_mass,
 95                charge_mass,
 96                chamber_volume,
 97                loss_fraction,
 98                start_pressure,
 99                reduced_burnrate,
100            ) = (float(e.get()) for e in self.value_entries[1:])
101
102            cross_section *= 1e-2  # dm^2 to m^2
103            chamber_volume *= 1e-3  # L to m^3
104            start_pressure *= 1e6  # MPa to Pa
105            loss_fraction *= 1e-2  # % to 1
106
107            charge = Charge.from_propellant(
108                reduced_burnrate=reduced_burnrate,
109                propellant=self.get_prop(),
110                form_function=self.form_function_frame.get_form_function(),
111            )
112
113            self.gun = Gun(
114                name=name,
115                description=self.text.get("1.0", "end-1c"),
116                cross_section=cross_section,
117                shot_mass=shot_mass,
118                charge=charge,
119                charge_mass=charge_mass,
120                chamber_volume=chamber_volume,
121                loss_fraction=loss_fraction,
122                start_pressure=start_pressure,
123            )
124            self.destroy()
125
126        except (ValueError, IndexError) as e:
127            logger.error(e)

239    def get_form_function(self) -> Optional[FormFunction]:
240        tab_index = self.notebook.index(self.notebook.select())
241
242        # try:
243        if tab_index == 0:
244            length, width, height = (float(e.get()) for e in self.non_perf_entries)
245            return FormFunction.non_perf(length=length, width=width, height=height)
246
247        elif tab_index == 1:
248            arch_width, height = (float(e.get()) for e in self.single_perf_entries)
249            return FormFunction.single_perf(arch_width=arch_width, height=height)
250
251        elif tab_index == 2:
252            arch_width, perforation_diameter, height = (
253                float(e.get()) for e in self.multi_perf_entries
254            )
255            return FormFunction.multi_perf(
256                arch_width=arch_width,
257                perforation_diameter=perforation_diameter,
258                height=height,
259                shape=self.get_shape(),
260            )
261
262        return None
263
264    def get_shape(self):
265        i = self.multi_perf_combo.current()
266        if i == -1:
267            raise ValueError("no shape selected")
268        else:
269            return self.multi_perf_shapes[i]
