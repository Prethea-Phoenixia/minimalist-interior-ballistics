 36    @staticmethod
 37    def remark(old_state: State, new_significance: Significance) -> State:
 38        return State(

 47    @cached_property
 48    def volume_burnup_fractions(self) -> tuple[float, ...]:
 49
 50        return tuple(
 51            charge.psi(max(min(burnup_fraction, charge.Z_k), 0))
 52            for charge, burnup_fraction in zip(self.gun.charges, self.burnup_fractions)
 53        )
 54
 55    @cached_property
 56    def average_pressure(self) -> float:
 57        """the length-averaged pressure in the equivalent gun, under the Lagrange
 58        gradient of 0-dimensional interior ballistics.
 59
 60        Notes
 61        -----
 62        The equivalent gun is a is formed by stretching the chamber volume of the
 63        actual gun into a section such that the gun is of uniform cross section
 64        throughout. The error of this treatment is usually insignificant for the range
 65        of conditions encountered in conventional firearms (although it is of more
 66        concern in light-gas guns).
 67        """
 68        l, v = self.travel, self.velocity
 69        l_psi = self.gun.l_0 * (1 - self.gun.incompressible_fraction(self.volume_burnup_fractions))
 70        if l_psi <= 0:
 71            return inf
 72        else:
 73            return self.gun.gas_energy(psis=self.volume_burnup_fractions, v=v) / (self.gun.S * (l_psi + l))
 74
 75    @cached_property
 76    def shot_pressure(self) -> float:
 77        """the shot-base pressure in the equivalent gun. For more info refer to
 78        `average_pressure`.
 79        """
 80        return self.average_pressure / (
 81            1 + self.gun.charge_mass_sum / (3 * self.gun.shot_mass * (1 + self.gun.loss_fraction))
 82        )
 83
 84    @cached_property
 85    def breech_pressure(self) -> float:
 86        """the breech face pressure in the equivalent gun. For more info refer to
 87        `average_pressure`.
 88        """
 89        return self.shot_pressure * (
 90            1 + self.gun.charge_mass_sum / (2 * self.gun.shot_mass * (1 + self.gun.loss_fraction))
 91        )
 92
 93    @cached_property
 94    def is_burnout(self) -> bool:
 95        return all(Z > charge.Z_k for charge, Z in zip(self.gun.charges, self.burnup_fractions))
